! CSS TIP !
1. 이미지를 가로로 연결하기 위해 포지션을 relative로 설정하고 영역에서 벗어나는 이미지를 안보이게 처리하기 위해 overflow: hidden을 설정해줍니다.
2. 이미지가 연속적으로 움직일 수 있도록 첫번째 이미지를 6번째에 추가로 넣습니다. (애니메이션 설정은 스크립트 파트에서!!)
3. 마찬가지로 이미지가 6장이기 때문에 inner의 가로 크기를 +하여 4800px로 설정합니다.
4. 닷 메뉴 스타일을 지정해줍니다.

#2. JAVASCRIPT : 스크립트 짜기

반복문을 통해 문자열을 생성하고 각 함수를 만들어 재활용이 가능하도록 합니다. 

{
    const sliderWrap = document.querySelector(".slider__wrap");
    const sliderImg = document.querySelector(".slider__img");   //보여지는 영역
    const sliderInner = document.querySelector(".slider__inner"); //움직이는 영역
    const slider = document.querySelectorAll(".slider");      //각각의 이미지
    const sliderBtn = document.querySelector(".slider__btn"); //버튼
    const sliderBtnPrev= sliderBtn.querySelector(".prev"); // 왼쪽버튼
    const sliderBtnNext = sliderBtn.querySelector(".next"); // 오른쪽 버튼
    const sliderDot = document.querySelector(".slider__dot"); // 닷메뉴

    let currentIndex = 0;                       //현재 이미지
    let sliderCount = slider.length;            //이미지 갯수
    let sliderWidth = sliderImg.offsetWidth;    //이미지 가로값

    let dotIndex = "";
    
    function intit(){
        // dot 만들어주기!
        // slider.forEach(() => {
        //     dotIndex += "<a href='#' class='dot'>이미지1</a>";
        // });
        // dot 만들어주기!
        slider.forEach(() => dotIndex += "<a href='#' class='dot'>이미지1</a>");
        sliderDot.innerHTML = dotIndex;

        //첫번째 닷 버튼 활성화 표시(active)
        sliderDot.firstChild.classList.add("active");

    }
    intit();

    //이미지 이동 스크립트
    function gotoSlider(num){
        sliderInner.style.transition = "all 400ms";
        sliderInner.style.transform = "translateX(" + -sliderWidth * num + "px)";
        currentIndex = num;


        //두번재 이미지에서 두번째 닷 버튼 활성화 (탭 메뉴와 원리 동일)
        //1. 닷 메뉴의 클래스를 모두 삭제
        //2. 해당 이미지의 해당 닷 메뉴에 클래스 추가
        let dotActive = document.querySelectorAll(".slider__dot .dot");
        dotActive.forEach(el => el.classList.remove("active"));
        dotActive[num].classList.add("active");
    }

    //버튼을 클릭할 때 이미지 이동 작동시키기
    document.querySelectorAll(".slider__btn a").forEach((btn, index) => {
        btn.addEventListener("click", () => {
            let prevIndex = (currentIndex + (sliderCount - 1 )) % sliderCount;
            let nextIndex = (currentIndex + 1) % sliderCount;

            if(btn.classList.contains("prev")){
                gotoSlider(prevIndex);
            }else{
                gotoSlider(nextIndex);
            }
        });
    });

    //dot 메뉴를 클릭할 때 이미지 이동 작동시키기
    document.querySelectorAll(".slider__dot .dot").forEach((dot, index) => {
        dot.addEventListener("click", () => {
            gotoSlider(index);
        });
    });

    //자바스크립트는 순서댜로 진행되므로 실행문이 맨 밑에 있어서 나오지 않은 것.
}
