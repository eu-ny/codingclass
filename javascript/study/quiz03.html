퀴즈 주관식(여러문제) 유형

자바스크립트를 이용하여 웹디자인기능사 필기문제를 주관식 유형 퀴즈로 여러개 제작해보기
4개의 주관식 퀴즈를 하드 출력, 반복문과 조건문을 이용한 출력으로 나누어 살펴보자!

#1. 선택자 자바스크립트 소스

상수에게 해당명칭(ex: quizType)을 선언해주고 쿼리셀랙터(태그, 아이디, 클래스를 선택할 수 있는 것)로 선택한 요소를 찾아서 넣어보자!

//선택자
const quizType = document.querySelectorAll(".quiz__type"); //퀴즈종류
const quizNumber = document.querySelectorAll(".quiz__question .number"); //퀴즈 번호
const quizAsk = document.querySelectorAll(".quiz__question .ask"); //퀴즈 질문
const quizConfirm = document.querySelectorAll(".quiz__answer .confirm"); //정답 확인 버튼
const quizResult = document.querySelectorAll(".quiz__answer .result"); //정답 결과
const quizInput = document.querySelectorAll(".quiz__answer .input"); //사용자 정답 작성란
const quizView = document.querySelectorAll(".quiz__view"); //강아지

=> 퀴즈 문제 만들기 02번에서는 문제가 하나였기 때문에 querySelector만 사용해주었지만 이번 스크립트는 퀴즈 문제가 여러개이므로 querySelector 뒤에 All을 붙여서 모든 선택자를 불러오도록 한다.

for(let i=0; i<quizInfo.length; i++ ){

#4. 정답 숨기기 자바스크립트 소스

반복문을 이용하여 총 4가지 문제의 정답을 숨겨보자! 

// 정답숨기기
// 일반 출력으로 정답 숨기기
quizResult[0].style.display = "none";
quizResult[1].style.display = "none";
quizResult[2].style.display = "none";
quizResult[3].style.display = "none";


// for문으로 정답 숨기기
for(let i=0; i<quizInfo.length; i++){
    quizResult[i].style.display = "none";
}


// forEach문으로 정답 숨기기
// forEach는 e > i > a 순서로 식을 읽는다.
// 즉 데이터에서 인덱스를 뽑고 싶다면 e 와 i를 모두 선언해주어야 출력이 가능하다. 만약 i만 선언해도 i를 e로 생각하여 읽는다.
quizInfo.forEach((e, i)=>{
    quizResult[i].style.display = "none";
});
정리
quizResult[i].style.display = "none";
각 인덱스의 Result값에 css스타일 중 display = "none";을 주어서 정답 결과 박스를 보이지 않게 해준다.

#5. 정답 확인하기 자바스크립트 소스

forEach문과 addEventListener를 통해 정답을 확인해보자! 그리고 classList를 통해 답에 따른 표정 변화를 선언해보자! 

// 정답확인하기
// quizConfirm.addEventListener("click",() => {});
// quizConfirm.addEventListener("click",function(){});
// quizConfirm를 클릭할 시 이벤트가 실행되도록한다.

quizConfirm.forEach((btn, num)=>{
    btn.addEventListener("click", ()=>{
        
        // 사용자 정답
        const userWord = quizInput[num].value; //사용자의 정답을 소문자로 변환시키고 여백을 없애준다.
        
        // 사용자 정답 = 문제 정답
        if(userWord == quizInfo[num].answerResult){ 
            // num = 특별한 기능으로 넣은 것이 아니라 인덱스를 표현하기 위한 명칭으로 이해하면 된다. ( 0번 객체의 퀴즈 1번 객체의 퀴즈 등 )
            // 정답
            quizView[num].classList.add("like");
            quizConfirm[num].style.display = "none";
        } else {
            // 오답
            quizView[num].classList.add("dislike");
            quizConfirm[num].style.display = "none";
            quizResult[num].style.display = "block";
            quizInput[num].style.display = "none"
            // 반복문을 사용할 경우 정답과 오답에 각각 스타일 선언을 따로 해주어야 한다.
        }
    });
});

forEach정리
1. quizConfirm의 수가 1개가 아닌 4개로 복수이기 때문에 forEach를 통해 문제가 총 4개라는 정보를 선언 해주어야 정상으로 이벤트 실행이 가능하다.
2. quizConfirm.addEventListener("click",(btn, num) => {});에서 btn == element / >num == index이며 이름을 붙여준 것 뿐, 특별한 기능이 가미된 것은 아니다.
3. quizConfirm == botten이며 상수에[num]을 붙여 저장한 데이터를 인덱스 별로 불러오게끔 한다.

데이터 저장 방법 : 배열 속 객체 타입
각 다른 여러개의 데이터를 저장할 때 효율젹인 방법이다. 

const quizInfo = [
    {키1 : 값1},{키2 : 값2},{키3 : 값3},{키4 : 값4}
];
//객체는 항상 키와 값(key-value pair)으로 값을 넣어야 하며 그 값들은 속성(Property)라고 부른다.
//객체 한쌍의 구분은 쉼표(comma)로 해준다.

for문을 통한 출력 방법
출력할 때 주의할 점은 데이터를 불러올 경우 저장 방법이 객체이기 때문에 불러오려는 데이터값의 키를 구체적으로 선언해주어야 한다.
즉 .answerType과 같은 키를 선택하지 않은 채로 출력하게 되면 무엇을 출력하라는 건지 인식하지 못하여 object로 값이 출력되는 오류가 난다. 

for(let i=0; i<4; i++ ){
    quizType[i].textContent = quizInfo[i].answerType; 
}

forEach을 통한 출력 방법
객체의 인덱스 별로 들어있는 해당 데이터를 뽑기 위한 것이기 때문에 e와 i를 선언해주어야 하며, 상수에도 똑같이 인덱스를 붙여주어야 오류가 생기지 않는다. 

quizInfo.forEach(function(e, i){
    quizType[i].textContent = quizInfo[i].answerType;
});

quizConfirm를 클릭할 시 이벤트가 실행되도록한다.

quizConfirm.addEventListener("click",function(){});

4개의 quizConfirm가 있음을 알려주고 버튼들의 이벤트가 실행될 수 있도록 한다.

quizConfirm.forEach((btn, num)=>{
    btn.addEventListener("click", ()=>{});
});
