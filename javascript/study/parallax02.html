#2. JAVASCRIPT : 스크립트로 페럴랙스 효과 주기

먼저 전체적인 흐름을 보고 스크립트가 어떻게 작동되는지 확인해봅시다! 🥹

💡 전체 스크립트 보기 
document.querySelectorAll("#parallax__dot a").forEach(el => {
    el.addEventListener("click", e => {
        e.preventDefault();

        document.querySelector(el.getAttribute("href")).scrollIntoView({behavior: "smooth"}); //scrollIntoView : 스크롤 이동 속성
        });


        window.addEventListener("scroll",() => {
        let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;

        document.querySelector("#parallax__info span").innerText = Math.floor(scrollTop);

        document.querySelectorAll(".content__item").forEach((e,i) => {
            if(scrollTop >= e.offsetTop - window.innerHeight/2){ //- window.innerHeight/2 브라우저 화면의 반 한 화면의(vh) 스크롤
                document.querySelectorAll("#parallax__dot li").forEach(li => {
                    li.classList.remove("active");
                    //해당되는 것들만 선택하여 순서대로 실행
                    document.querySelector("#parallax__dot li:nth-child("+(i+1)+")").classList.add("active");
                });
            }
        });
    });
});

❗️ 그럼 다음의 코드를 보며 함께 스크립트를 알아가봅시다 🥹 

//1 #parallax__dot의 모든 a요소 선택하기
document.querySelectorAll("#parallax__dot a").forEach(el => {
    //2 클릭시 해당 요소의 이벤트 속성 없애기
    el.addEventListener("click", e => {
      e.preventDefault();

    //3 새로운 요소의 속성과 이벤트 속성을 부여하기
      document.querySelector(el.getAttribute("href")).scrollIntoView({behavior: "smooth"}); //scrollIntoView : 스크롤 이동 속성
    });
});

1. 먼저 모든 메뉴(dot)을 선택해주어야 하므로 querySelectorAll을 통해 해당 html을 선택해줍니다.

2. 반복문을 통해 모든 메뉴의 a를 클릭시 e.preventDefault()를 실행하여 a가 가지고 있는 이벤트 속성(링크 이동)을 막아줍니다.

3. querySelector에서 요소의 속성을 불러오는 getAttribute을 사용하여 href의 이동 이벤트 선택한 뒤 scrollIntoView를 사용하여 스크롤 이동 시 부드럽게 이벤트가 작동되도록 합니다. 

//1 스크롤값 가져오기 (저장)
window.addEventListener("scroll",() => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;

    //2 스크롤 값 불러오기 (출력)
    document.querySelector("#parallax__info span").innerText = Math.floor(scrollTop);

    //3 선택한 버튼에게만 active 선택자를 부여하여 스타일 주기
    document.querySelectorAll(".content__item").forEach((e,i) => {
        if(scrollTop >= e.offsetTop - window.innerHeight/2){             //- window.innerHeight/2 : 브라우저 화면의 1/2(반) 스크롤(= 2vh)
            document.querySelectorAll("#parallax__dot li").forEach(li => {
                li.classList.remove("active");
                document.querySelector("#parallax__dot li:nth-child("+(i+1)+")").classList.add("active");
            })
        }
    });
});

1. 브라우저에게 이벤트리스너를 통해 스크롤을 할 때 이벤트를 설정합니다. 이때 스크롤의 현재 위치값을 가져오기 위해서 변수에게 window.pageYOffset, window.scrollY, document.documentElement.scrollTop을 || 연산자를 통해 모두 저장해줍니다. 이렇게 3개의 속성값을 모두 불러오는 이유는 다양한 환경의 브러우저에서도 거의 동일한 스크롤 값을 불러올 수 있기 때문입니다.

2. 그 뒤 querySelector를 통해 변수에 저장한 스크롤 값을 대입할 html을 선택합니다. 그리고 innerText를 통해 변수를 대입시킵니다. (이때 값을 깔끔하게 처리하기 위하여 수학 메서드인 Math.floor를 사용하여 소숫점을 반올림하도록 설정합니다.)

3. 만약 scrollTop(현재 스크롤값)이 e.offsetTop(각 섹션의 스크롤 값) - window.innerHeight/2(브라우저 화면의 1/2(반) 스크롤)보다 크거나 같을 경우 모든 li(버튼)의 active선택자를 지우고 선택된 li(버튼)에게만 active선택자를 선언해줍니다.
