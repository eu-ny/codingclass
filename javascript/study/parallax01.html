🖱 페럴렉스 효과 : 메뉴 스크롤링 효과 🖱

페럴랙스(parallax) 시차란 의미로 사용자가 스크롤링을 할 때 배경과 요소마다 속도 및 크기의 차이를 주어 입체적으로 느껴지게 하는 효과를 말합니다.
이번 시간에는 버튼을 눌렀을 때(active) 해당 화면으로 스크롤이 부드럽게 이동하는 효과와 스크롤을 내릴 때 해당 버튼이 강조되는 효과를 실습해보겠습니다. 😆


#1. HTML / CSS 설정하기

! HTML TIP !
각 섹션 별로 구조를 설정합니다. 이 때 큰 틀부터 구조를 설정하는 것이 중요한 포인트 입니다!
구조는 먼저 크게 nav : 메뉴 | main : 메인 콘텐츠(이미지, 텍스트) | aside : 스크롤의 위치값으로 나누어 줍니다.
세부적인 구조는 아래의 코드를 참고하시기 바랍니다!

! JAVASCRIPT TIP !
1. .scrollIntoView : 특정위치로 화면을 이동시키는 메서드 : 즉 버튼을 눌렀을때 원하는 내용이 있는 위치로 이동시켜야 할 때 사용할 수 있다.
2. element.getAttribute() : 선택한 요소의 특정 속성 값을 가져오는 메서드
3. 스크롤 수직 위치값을 불러오는 3가지 방법 : pageYOffset / scrollY / scrollTop
4. event.preventDefault : 일반적으로 브라우저에 의해 처리되는 기존의 액션(동작) 즉 이벤트 발생을 막아준다. ex) a 태그의 기본 이벤트인 href 링크 이동을 막아서 이동하지 않게 설정. 

 const scroll = document.querySelector("#parallax__info .scroll span");

window.addEventListener("scroll", () => {
    //스크롤 수직 위치값을 불러오는 3가지 방법 : pageYOffset / scrollY / scrollTop : 3가지 속성을 모두 사용해야 모든 브라우저에서 위치값을 가져올 수 있다.
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    
    //forEach문 - active 없애고 첫번째에게 다시추가 시켜 추가된 요소만 효과가 나타나도록 하기
    //스크롤 위치 값이 요소의 offsetTop(요소의 수직 값) 값보다 크거나 같을 경우 
    document.querySelectorAll(".content__item").forEach((el, index) => {
    if (scrollTop >= el.offsetTop -2) { //현 페이지에서 스크롤이 마지막 요소값에 미치지 못하는 오류가 있어 오차를 줄이기 위해 -2를 해주어서 오류를 완화해주었다.
        document.querySelectorAll("#parallax__nav li").forEach((li) => {
        li.classList.remove("active"); //li의 전체를 선택하기 위해 forEach문을 통해 li 값을 뽑아내고 li 요소에 선언되어있는 모든 active 선택자를 지워준다.
        });
        //위 조건이 실행된 후 다시 li 요소의 첫번째에게 class로 active를 선언한다.
        //인덱스에 1을 추가한 이유는 인덱스는 0부터 시작하기 때문. 즉 0부터 시작하면 마지막 영역을 포함하지 못하므로 1부터 시작하기 위해 맞춰준 것
        document.querySelector("#parallax__nav li:nth-child(" + (index + 1) + ")").classList.add("active");
    }
    });

    //info
    //스크롤 위치값이 소숫점으로 나오기 때문에 소숫점을 올림 및 반올림을 해주어 값이 딱 떨어지게 해준다
    document.querySelector(".scroll span").innerText = Math.ceil(scrollTop);    //Math.ceil = 값을 올려서 숫자로 반환시키는 

    //for()
    //선택자 offset에게 section의 n번째 offsetTop값을 innerText를 통해 저장시킨다.
    for (let i = 1; i <= 9; i++) {
    document.querySelector(".offset" + i).innerText = document.getElementById("section" + i).offsetTop;
    }

    //스크롤 이동
    // 링크를 클릭했을 때 딱딱한 클릭 이벤트가 실행되지 않고 새로 부드러운(smooth) 이벤트 만들어주기
    document.querySelectorAll("#parallax__nav li a").forEach((li) =>{
        li.addEventListener("click", (e) =>{
            e.preventDefault(); // event.preventDefault : 클릭했을 때 li가 가지고 있던 a의 속성인 href(링크)를 실행시키지 않는다.
            document.querySelector(li.getAttribute("href")).scrollIntoView({ 
            behavior: "smooth", //부드럽게 움직이도록 : 스크롤이 움직일 때.
            });
        });
    });
});
