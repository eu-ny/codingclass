//선택자
const cssList = document.querySelector(".search__list .line");

//데이터 출력하기
const updateList = () => {
    let listCSS = "";

    cssProperty.forEach((data, index) => {
        listCSS += `
  • ${data.num}. ${data.name} : ${data.desc}
    • `
    });

    cssList.innerHTML = listCSS;
}
updateList();

//스크립스는 순서대로 작동하므로 순서 주의

//반대로 정렬
function sortReverse(){
    cssProperty.reverse();
    updateList();
}
//내림차순 정렬
function sortDescending(){
    cssProperty.sort((a,b) => {
        return b.num - a.num;
    });
    updateList();
}
//오름차순 정렬
function sortAscending(){
    cssProperty.sort((a,b) => {
        return a.num - b.num;
    });
    updateList();
}



//미션!
//알파벳(a-z) 정렬
function sortAlphabet(){
    cssProperty.sort((a,b) => {
        return a.name.localeCompare(b.name);
    });
updateList();
}

//알파벳(z-a) 정렬
function sortAlphabetZ(){
    cssProperty.sort((a,b) => {
        return b.name.localeCompare(a.name);
    });
updateList();
}
